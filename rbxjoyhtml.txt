    function goToStep(step) {

    function generateRobuxCounter() {
      const amountSelect = document.getElementById('amount');
      const amount = parseInt(amountSelect.value);
      let counter = 0;
      const robuxCounter = document.getElementById('robuxCounter');
      const completeMsg = document.getElementById('generationCompleteMsg');

      robuxCounter.textContent = "0 Robux";
      completeMsg.style.display = 'none';

      const stepAmount = Math.ceil(amount / 100);
      const interval = setInterval(() => {
        counter += stepAmount;
        if (counter >= amount) {
          counter = amount;
          clearInterval(interval);
          completeMsg.style.display = 'block';
          // بعد 1.5 ثانية ننتقل للخطوة 3 (شريط التحميل)
          setTimeout(() => {
            goToStep(3);
          }, 1500);
        }
        robuxCounter.textContent = counter.toLocaleString() + " Robux";
      }, 30);
    }
